
Mnemo has been deployed to your Kubernetes cluster.

{{- $fullname := include "mnemo.fullname" . -}}

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ $fullname }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ .Release.Namespace }} svc {{ $fullname }} -w
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ $fullname }} 8080:{{ .Values.service.port }}
  echo "Visit http://localhost:8080"
{{- end }}

2. Verify the deployment is healthy:
  kubectl --namespace {{ .Release.Namespace }} get pods -l "app.kubernetes.io/name={{ include "mnemo.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. Check the health endpoint:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ $fullname }} 8080:{{ .Values.service.port }}
  curl http://localhost:8080/health

4. Access the API documentation:
  # Swagger UI:  http://localhost:8080/swagger-ui
  # OpenAPI JSON: http://localhost:8080/api/v1/openapi.json

5. Access gRPC (same port):
  grpcurl -plaintext localhost:8080 list

{{- if .Values.ingress.enabled }}
{{- if not (gt (len .Values.ingress.tls) 0) }}

WARNING: Ingress is enabled without TLS!
  Traffic between clients and the ingress controller will be unencrypted.
  Configure TLS in your values:

    ingress:
      tls:
        - secretName: mnemo-tls
          hosts:
            - {{ (index .Values.ingress.hosts 0).host }}

  With cert-manager:
    ingress:
      annotations:
        cert-manager.io/cluster-issuer: letsencrypt-prod
      tls:
        - secretName: mnemo-tls
          hosts:
            - {{ (index .Values.ingress.hosts 0).host }}

{{- end }}
{{- end }}

{{- if not .Values.mnemo.bootstrapKeys }}
{{- if not .Values.mnemo.existingSecret }}

WARNING: No authentication keys configured!
  Create a Kubernetes Secret and reference it via mnemo.existingSecret:

    kubectl create secret generic mnemo-auth \
      --from-literal=MNEMO_AUTH_API_KEYS="$(openssl rand -base64 32)"
    helm upgrade {{ .Release.Name }} mnemo --set mnemo.existingSecret=mnemo-auth

  SECURITY: Do NOT pass secrets via --set as they are stored in Helm release metadata.
{{- end }}
{{- end }}

{{- if .Values.qdrant.enabled }}
{{- if not .Values.qdrant.apiKey }}

NOTE: Qdrant is deployed without API key authentication.
  Qdrant is accessible without credentials from any pod in the namespace.
  For production, enable Qdrant auth and pass the key to Mnemo:

    qdrant:
      apiKey: true   # auto-generates a key stored in a Secret

    mnemo:
      extraEnv:
        - name: MNEMO_QDRANT_API_KEY
          valueFrom:
            secretKeyRef:
              name: {{ .Release.Name }}-qdrant-apikey
              key: api-key

  Or use a NetworkPolicy to restrict access (--set networkPolicy.enabled=true).
{{- end }}
{{- end }}
